Classificatore basato su Support Vector Machine (SVM) lineari e informazioni linguistiche non lessicali.

Il primo classificatore dovr essere costruito utilizzando SVM lineari, impiegando come rappresentazione del testo esclusivamente informazioni linguistiche non lessicali, estratte tramite il sistema Profiling-UD. Per questo modello, richiesto di:

  • effettuare una valutazione tramite procedura di cross-validation a 5 fold sul training set;
  • valutare le prestazioni del modello sui set di validazione e di test;
  • fornire un elenco delle 20 feature pi rilevanti ai fini della classificazione.

# Creazione dizionario con le features 
# Nomi file nel csv: Dataset/Dataset/Test/colombi_101_test.conllu

def carica_features(path):
    dizionario_featuresUD = {}
    nomi_features = None
    
    with open(path, "r") as file:
        file_ud = csv.reader(file, delimiter="\t")
        for row in file_ud:
            if nomi_features is None:
                nomi_features = row[1:] # salvo la prima riga con i nomi delle features
            
            else:
                # di ogni riga salvo l'id, il nome dell'autrice e lo split
                nome_file = row[0].split("/")[-1]
                doc_id = nome_file.split(".")[0]
                nome_autrice = doc_id.split("_")[0]
                split = doc_id.split("_")[-1]

                features = [float(el) for el in row[1:]] # salvo le features del documento

                # inserisco i dati nel dizionario
                dizionario_featuresUD[doc_id] = {'autrice': nome_autrice, 'split': split, 'features': features}

    return dizionario_featuresUD, nomi_features

train_features, test_features, validation_features = [], [], []
train_autrici, test_autrici, validation_autrici = [], [], []

for doc_id in dataset:
    split = dataset[doc_id]['split']
    features = dataset[doc_id]['features']
    autrice = dataset[doc_id]['autrice']

    if split == "training":
        train_features.append(features)
        train_autrici.append(autrice)

    elif split == "test":
        test_features.append(features)
        test_autrici.append(autrice)
    
    else: # if split == "validation"
        validation_features.append(features)
        validation_autrici.append(autrice)

In[22]:
all_y_true = []
all_y_pred = []

for i in range(len(folds)):
    train_ids = folds[i][0]
    test_ids = folds[i][1]
    

    fold_X_train = X_train[train_ids]
    fold_y_train = y_train[train_ids]

    fold_X_test = X_train[test_ids]
    fold_y_test = y_train[test_ids]

    # Scaliamo le features
    scaler = MinMaxScaler()
    fold_X_train = scaler.fit_transform(fold_X_train)
    fold_X_test = scaler.transform(fold_X_test)

    kfold_svc = LinearSVC(dual=False, max_iter=10000)
    kfold_svc.fit(fold_X_train, fold_y_train)
    fold_y_pred = kfold_svc.predict(fold_X_test)
    fold_accuracy = accuracy_score(fold_y_test, fold_y_pred)

    dummy_clf = DummyClassifier(strategy="most_frequent")   # dummy classifier viene utilizzato per avere una baseline
    dummy_clf.fit(fold_X_train, fold_y_train)
    dummy_score = dummy_clf.score(fold_X_test, fold_y_test)
    
    all_y_true += fold_y_test.tolist()
    all_y_pred += fold_y_pred.tolist()
    print(f"Accuracy fold {i+1}: {fold_accuracy}, baseline: {dummy_score}")
print(f"\nOverall accuracy: {accuracy_score(all_y_true, all_y_pred)}")
